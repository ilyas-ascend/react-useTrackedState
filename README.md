# react-usetrackedstate

## How to use

### Installation

```shell
$ npm i react-usetrackedstate
```
```shell
$ yarn add react-usetrackedstate
```

```javascript
import useTrackedState from 'react-usetrackedstate'
function MyComponent(){
  const counter = useTrackedState({ value: 0 })

  useEffect(() => {
    setInterval(() => {
      counter.value++;
    }, 1000);
  }, [])

  return (
      <p>{counter.value}</p>
  ) 
}
```
